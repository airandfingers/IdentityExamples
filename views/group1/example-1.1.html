Sample code taken from Bootstrap

<!-- taken from http://getbootstrap.com/examples/signin/ -->

<!-- Note, there is no METHOD defined for the form, so browser will default to GET -->
  <form class="form-signin">
    <h2 class="form-signin-heading">Please sign in</h2>
    <!-- Note, the form parameters are not named -->
    <input type="text" class="form-control" placeholder="Email address" name="username" required autofocus>
    <input type="password" class="form-control" placeholder="Password" name="password" required>
    <label class="checkbox">
      <input type="checkbox" value="remember-me"> Remember me
    </label>
    <button class="btn btn-lg btn-primary btn-block" type="submit">Sign in</button>
  </form>
